Production Code Stack

Next.js 16 + OpenRouter AI + Razorpay

See real production code integrating OpenRouter free AI keys with Next.js 16 Server Actions and Razorpay payments.

import { NextRequest, NextResponse } from 'next/server';
import Razorpay from 'razorpay';

// Next.js 16 Server Action with OpenRouter API + Razorpay
export async function POST(req: NextRequest) {
  const { prompt } = await req.json();
  
  // Call Free OpenRouter AI API (DeepSeek R1 / Llama 3)
  const aiRes = await fetch('https://openrouter.ai/api/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Authorization': \`Bearer \${process.env.OPENROUTER_API_KEY}\`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'deepseek/deepseek-r1:free',
      messages: [{ role: 'user', content: prompt }]
    })
  });

  return NextResponse.json(await aiRes.json());
}
